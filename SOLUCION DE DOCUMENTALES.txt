    const words = fullNarration.split(/\s+/).filter(Boolean);
    const totalDurationSec = duration * 60;
    const secPerWord = totalDurationSec / Math.max(words.length, 1);
    const chunkSize = 10;
    let srtContent = '';
    let srtIndex = 1;
    for (let i = 0; i < words.length; i += chunkSize) {
      const chunk = words.slice(i, i + chunkSize).join(' ');
      const startSec = i * secPerWord;
      const endSec = Math.min((i + chunkSize) * secPerWord, totalDurationSec);
      const fmt = (s) => {
        const h = Math.floor(s / 3600).toString().padStart(2, '0');
        const m = Math.floor((s % 3600) / 60).toString().padStart(2, '0');
        const sec = Math.floor(s % 60).toString().padStart(2, '0');
        const ms = Math.round((s % 1) * 1000).toString().padStart(3, '0');
        return `${h}:${m}:${sec},${ms}`;

    setStage('assembly', 'running');

    const validImages = imagePaths.filter(p => p && fs.existsSync(p));

    let ffmpegResult;

    if (validImages.length > 0) {
      // ── Montaje REAL: imágenes + narración + subtítulos ──

      // Crear lista de imágenes para FFmpeg
      const imgListPath = path.join(outputDir, `${jobId}_images.txt`);

        '-pix_fmt', 'yuv420p',
        '-movflags', '+faststart',
        videoPath

      ffmpegResult = await new Promise((resolve, reject) => {
        const proc = spawn('ffmpeg', ffmpegArgs);
        let stderr = '';
        proc.stderr.on('data', d => { stderr += d.toString(); });
        proc.on('close', code => {
          if (code === 0) resolve({ success: true });
          else reject(new Error(`FFmpeg salió con código ${code}:\n${stderr.slice(-1000)}`));
        });
      });

      // Limpiar archivos temporales
      try { fs.unlinkSync(imgListPath); } catch {}

    } else {
      // ── Fallback: video con fondo oscuro y narración ──
      const bgColor = visualStyle === 'cinematic' ? '0x0a0a0a' : '0x1a1a2e';
      const srtFilter = fs.existsSync(srtPath)
        ? `,subtitles='${srtPath.replace(/\\/g, '/').replace(/:/g, '\\:')}'`
        : '';

      const drawTextFilter = [
        `drawtext=text='${(scriptData.title || topic).replace(/'/g, "\\'").slice(0, 60)}'`,
        `fontcolor=white:fontsize=${Math.round(dims.w * 0.04)}:x=(w-text_w)/2:y=(h-text_h)/2`,
        `box=1:boxcolor=black@0.5:boxborderw=10`,
        `enable='between(t,0,5)'`
      ].join(':');

      const ffmpegArgs = [
        '-y',
        '-f', 'lavfi',
        '-i', `color=c=${bgColor}:s=${dims.w}x${dims.h}:r=25`,
        ...(audioPath && fs.existsSync(audioPath) ? ['-i', audioPath] : []),
        '-vf', `${drawTextFilter}${srtFilter}`,
        '-c:v', 'libx264', '-preset', 'fast', '-crf', '23',
        '-t', String(totalDurationSec),
        ...(audioPath && fs.existsSync(audioPath) ? ['-c:a', 'aac', '-b:a', '192k', '-shortest'] : []),
        '-pix_fmt', 'yuv420p',
        '-movflags', '+faststart',
        videoPath
      ];

      ffmpegResult = await new Promise((resolve, reject) => {
        const { spawn } = require('child_process');
        const proc = spawn('ffmpeg', ffmpegArgs);
        let stderr = '';
        proc.stderr.on('data', d => { stderr += d.toString(); });
        proc.on('close', code => {
          if (code === 0) resolve({ success: true });
          else reject(new Error(`FFmpeg salió con código ${code}:\n${stderr.slice(-1000)}`));
        });
      });
    }

    setStage('assembly', 'completed', { videoPath });

    // ── ETAPA 10: EXPORTACIÓN ─────────────────────────────
    setStage('export', 'running');
    const videoFileName = path.basename(videoPath);
    const videoUrl = `/uploads/documentary/${videoFileName}`;
    const videoExists = fs.existsSync(videoPath);
    const videoSize = videoExists ? fs.statSync(videoPath).size : 0;

    job.status = 'completed';
    job.progress = 100;
    job.result = {
      videoPath,
      videoUrl,
      videoFileName,
      videoSize,
      scriptPath,
      srtPath,
      audioPath,
      imagePaths: imagePaths.filter(Boolean),
      metadata: {
        topic, duration, resolution, format, language,
        narrativeStyle, visualStyle,
        scenesCount: scenesData.length,
        wordCount: words.length,
        createdAt: new Date().toISOString()
      }
    };
    setStage('export', 'completed', {
      videoUrl,
      videoSize: `${(videoSize / 1024 / 1024).toFixed(1)} MB`,
      downloadReady: videoExists
    });
    documentaryJobs.set(jobId, { ...job });
    console.log(`[Documentary] ✅ Job ${jobId} completado — ${videoUrl} (${(videoSize/1024/1024).toFixed(1)} MB)`);

  } catch (error) {
    console.error(`[Documentary] ❌ Job ${jobId} error:`, error.message);
    job.status = 'failed';
    job.error = error.message;
    job.stages[job.stage].status = 'failed';
    documentaryJobs.set(jobId, { ...job });
  }

    } else {

    documentaryJobs.set(jobId, { ...job });
